00095 // Note: The "if (!is.get(c)) ..." construct used below
00096 //       is needed to handle "unexpected EOF" situations.
00097 Token    *Token::nextToken(istream& is)
00098 {
00099     string  s;
00100     char    c = -1;
00101 
00102     // Worth the trouble doing anything ?
00103     if (!is)
00104         return new Token(Token::END);
00105 
00106     // Get one character (if possible)
00107     if (!is.get(c))
00108         return new Token(Token::END);
00109 
00110     // Skip any white space ...
00111     while (isBlank(c)) {
00112         if (!is.get(c))
00113             return new Token(Token::END);
00114     }
00115 
00116     // What have we got now ?
00117 
00118     switch(c) { // First look for special cases
00119 
00120     case '\n': case '\r':
00121         return new Token(Token::EOL);
00122 
00123     case ';':   // Can be either ; or ;;
00124         if (Token::level >= Token::SH) {    // iff sh-script: could be ;;
00125             if (!is.get(c)) // lookahead
00126                 return new Token(Token::SEQUENCE);
00127             if (c == ';')
00128                 return new Token(Token::BREAK);
00129             is.putback(c);  // undo lookahead
00130         }
00131         return new Token(Token::SEQUENCE);
00132 
00133     case '&':   // Can be either & or &&
00134         if (Token::level > Token::BASIC) {  // iff advanced: could be &&
00135             if (!is.get(c)) // lookahead
00136                 return new Token(Token::BACKGROUND);
00137             if (c == '&')
00138                 return new Token(Token::AND);
00139             is.putback(c);  // undo lookahead
00140         }
00141         return new Token(Token::BACKGROUND);
00142 
00143     case '|':   // Can be either | or ||
00144         if (Token::level > Token::BASIC) {  // if advanced: could be ||
00145             if (!is.get(c)) // lookahead
00146                 return new Token(Token::PIPE);
00147             if (c == '|')
00148                 return new Token(Token::OR);
00149             is.putback(c);  // undo lookahead
00150         }
00151         return new Token(Token::PIPE);
00152 
00153     case '>':   // Can be either > or >>
00154         if (!is.get(c)) // lookahead
00155             return new Token(Token::OUTPUT);
00156         if (c == '>')
00157             return new Token(Token::APPEND);
00158         is.putback(c);  // undo lookahead
00159         return new Token(Token::OUTPUT);
00160 
00161     case '<':   // Can be either < or <<
00162         if (Token::level >= Token::SH) {    // if sh-scripts: could be <<
00163             if (!is.get(c)) // lookahead
00164                 return new Token(Token::INPUT);
00165             if (c == '<')
00166                 return new Token(Token::HERE);
00167             is.putback(c);  // undo lookahead
00168         }
00169         return new Token(Token::INPUT);
00170 
00171     case '(':   // subshell begin
00172         if (Token::level == Token::BASIC)   // must at least be advanced level
00173             break;      // treat like ordinary char
00174         return new Token(Token::SUBS);
00175 
00176     case ')':   // subshell end
00177         if (Token::level == Token::BASIC)   // must at least be advanced level
00178             break;      // treat like ordinary char
00179         return new Token(Token::ENDS);
00180 
00181     case '$':   // variables: $something
00182         if (Token::level == Token::BASIC)   // must at least be advanced level
00183             break;      // treat like ordinary char
00184         if (!is.get(c)) // lookahead
00185             return new Token(Token::WORD, "");  // just a $ ?
00186         if (Token::level >= Token::BASH) {  // bash/ksh ?
00187             // The $( pair is not a variable
00188             if (c == '(')
00189                 return new Token(Token::EVAL);
00190         }
00191         // The following non-letter pairs are common
00192         //     $* $@ $# $0 ... $9 $? $! $$
00193         if (strchr("*@#0123456789?!$", c) != NULL) {
00194             s += c;
00195             return new Token(Token::VAR, s);
00196         }
00197         // Having handled all the special cases,
00198         // we now expect a true word (e.g. $PATH)
00199         // which MUST begin with a letter.
00200         if (!isalpha(c)) {
00201             is.putback(c);          // Can not use this one
00202             return new Token(Token::WORD, "");  // just a $ ?
00203         }
00204         // Oke, this looks promissing ...
00205         // Collect an alfanumeric string (a _ also counts as a letter)
00206         do {
00207             s += c;
00208             if(!is.get(c))
00209                 return new Token(Token::VAR, s);
00210         } while (isalnum(c) || (c == '_'));
00211         is.putback(c);  // undo lookahead
00212         return new Token(Token::VAR, s);
00213 
00214     case '{':   // sh-script ? group begin
00215         if (!(Token::level >= Token::SH))
00216             break;      // treat like ordinary char
00217         return new Token(Token::GROUP);
00218 
00219     case '}':   // sh-script ? group end
00220         if (!(Token::level >= Token::SH))
00221             break;      // treat like ordinary char
00222         return new Token(Token::GRPEND);
00223 
00224     case '`':   // sh-script? (back-quotes)
00225         if (!(Token::level >= Token::SH))
00226             break;      // treat like ordinary char
00227         return new Token(Token::BQUOTE);
00228 
00229     } // end special character switch
00230 
00231     // Assume it is just a word;
00232     // Read any text upto something special.
00233     Token::token_t  type = Token::WORD;
00234 
00235     do {
00236         s += c;
00237         if (!is.get(c))
00238             return new Token(type, s);
00239         if (Token::level > Token::BASIC) {  // advanced ?
00240             // An embedded (simple) wildcard character ?
00241             // MAYBE: Handle [...] as well
00242             if (strchr("*?", c) != NULL)    // yes
00243                 type = Token::EXPR;         // change type from WORD to EXPR
00244         }
00245     } while (!isSpecial(Token::level, c));
00246     is.putback(c);  // undo lookahead
00247 
00248     return new Token(type, s);
00249 }
